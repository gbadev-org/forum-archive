void CubePrint(char Text[12],float x,int y,float z){
   int Counter;
   char Char;

   glLoadIdentity();
   glTranslatef(0.0f+x,0.0f+y,0.0f+z);
   glRotatef(xrot,1.0f,0.0f,0.0f);
   glRotatef(yrot,0.0f,1.0f,0.0f);
   glRotatef(zrot,0.0f,0.0f,1.0f);
   
   for(Counter=0;Counter<=255;Counter++){
      Char=Text[Counter]-32;
      for(int xx=Char*4;xx<=Char*4+3;xx++)for(int yy=0;yy<=6;yy++){
      glTranslatef((xx-Char*4+Counter*5)*size*2,((yy*-1)*(size*2)),0);
      if(Font0[xx+yy*380]==2)AddQuat();
      glTranslatef(-(xx-Char*4+Counter*5)*size*2,-((yy*-1)*(size*2)),0);
      }
      if(Text[Counter+1]=='\0')Counter=255;
   }
}
