the following code fixed it:
Code:
void UpdateFog()
{
   const int32 Mass = 8;
   int32 density = 0;

   glFogColor(31, 0, 0, 15);
   glFogDepth(8);

   for(int8 i = 2; i <= 64; i += 2)
   {
      density = i * Mass;
      if(density > 127) density = 127;
      GFX_FOG_TABLE[(i - 2) / 2 ] = density;
   }
}
