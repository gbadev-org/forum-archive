if (a != b)
{
printf("Foo\n");
exit(0);
}
else
exit(0);

Code:

@ r0 = a, r1 = b
cmp r0, r1
beq 1f
stmfd sp!, {r0-r3, lr}
ldr r0, =textstring
BL printf
ldmfd sp!, { r0-r3, lr}
b 2
1:

2:


etc..
