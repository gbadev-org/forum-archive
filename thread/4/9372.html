{
    for(px = 0; px < x_res; px++)
    {
        x = x0 = px;
        y = y0 = py;

        x2 = x * x;
        y2 = y * y
 
        iter = 0;
        max_iter = 1000;
 
        while((x2 + y2) < (4) && (iter < max_iter))
        {
            y = 2 * x * y + y0;
            x = x2 - y2 + x0;

            x2 = x * x;
            y2 = y * y;

            iter++;
        }
 
        if(iter == max_iter)
        {
            color = 0;
        }
        else

            color = iter;
