Do something like this:
Code:
/* vblsleep() : wait for roughly vblanks/60 seconds */
int vblsleep(int vblanks)
{
  for(; vblanks > 0; vblanks--)
    VblankIntrWait();
}

int main(void)
{
  do_something();
  vblsleep(120);
  do_something_else();
  while(1) { }
}
